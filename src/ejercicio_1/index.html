  1. Pago por vehiculo = si vehiculo es tipo 1 es igual a 1

  2. Procesar vehiculo = si vehiculo es tipo 1 es igual a 1
                      suma uno al contador tipo 1
                    sino si vehiculo es tipo 2 es igual a 2
                      suma uno al contador tipo 2
                    sino si vehiculo es tipo 3 es igual a 3
                      suma uno al contador tipo 2

                    suma uno al contador de todos los vehiculos

                    acumula el pago del vehiculo que esta procesando
  3. Porcentaje de vehiculos tipo 1 = (cantidad de tipo 1 / cantidad total) * 100
  4. Porcentaje de vehiculos tipo 2 = (cantidad de tipo 2 / cantidad total) * 100
  5. Porcentaje de vehiculos tipo 3 = (cantidad de tipo 3 / cantidad total) * 100 
        
class Cl_Vehiculo {
  constructor(placa, tipoVehiculo) {
    this.placa = placa;
    this.tipoVehiculo = tipoVehiculo;
  }

  Pago() {
    if (this.tipoVehiculo === 1) {
      return 10;
    } else if (this.tipoVehiculo === 2) {
      return 20;
    } else if (this.tipoVehiculo === 3) {
      return 50;
    }
  }
}
         
      
        
class Cl_Peaje {
  constructor() {
    this.acumPagosTotal = 0;
    this.contTipo1 = 0;
    this.contTipo2 = 0;
    this.contTipo3 = 0;
    this.contTotal = 0;
  }

  ProcesarVehiculo(vehiculo) {
    if (vehiculo.tipoVehiculo === 1) {
      this.contTipo1++;
    } else if (vehiculo.tipoVehiculo === 2) {
      this.contTipo2++;
    } else if (vehiculo.tipoVehiculo === 3) {
      this.contTipo3++;
    }
    this.contTotal++;
    this.acumPagosTotal += vehiculo.Pago();
  }

  PorcentajeVehiculoTipo1() {
    return (this.contTipo1 / this.contTotal) * 100;
  }

  PorcentajeVehiculoTipo2() {
    return (this.contTipo2 / this.contTotal) * 100;
  }

  PorcentajeVehiculoTipo3() {
    return (this.contTipo3 / this.contTotal) * 100;
  }

  TotalPagos() {
    return this.acumPagosTotal * 0.4;
  }
}
